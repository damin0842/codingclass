CSS 문서를 SCSS를 이용하여 단순화 하였습니다.

  body {

  $dog-width: 100px;

  $color-gray: #eaebec;
  $easing: ease-in-out;
  
  //dog
  .dog{
    width: $dog-width;
    height: $dog-width;
    z-index: 1;
    
    &:before{
      content: '';
      display: block;

      height: 100%;

    position:absolute;

  .dog-body, .dog-head, .dog-torso{
    border-radius:50%;

    width:100%;
    height:100%;
  }
  // dog-body
  .dog-body{
    top:-50%;
    box-shadow: inset 0 -15px 0 0 #888;
    animation: dog-body $interval $easing infinite alternate;
    
    &:before{
      content:'';
      position:absolute;
      bottom: 90%;
      right: 50%;
      width:90%;
      height:90%;

      animation: dog-tail-blur $interval $interval / 6 $easing infinite alternate both;
      
      @keyframes dog-tail-blur{
        0%{
          transform:rotate(0);
          opacity:0;
        }
        50%{
          opacity:1;
        }
        100%{
          transform:rotate(90deg);
          opacity:0;
        }
      }
    }
    
    @keyframes dog-body {
      from{transform:translatex(-10%)}
      to{transform:translatex(10%)}
    }
  }
  
  //dog-head

      left:100%;

    }

  .dog-tongue{
    width: 40%;

        0%{transform: rotate(-20deg)}
        100%{transform: rotate(20deg)}
      }
    }
  }
  .dog-body > .dog-tail{
    bottom:93%;
  }
